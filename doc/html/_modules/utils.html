[docs]    def set_request_params(self,cs):
        """Set the CoordinateSystemParameters values by ROS service request (CsRefPoints.srv).
        
        Args:
            cs (object): struct with the values of the parameters, stated by the ROS service call
        """
        self.name         = cs.name_cs.data
        self.d            = cs.distance.data
        self.x1           = cs.p1.x
        self.y1           = cs.p1.y
        self.x2           = cs.p2.x
        self.y2           = cs.p2.y
        self.x3           = cs.p3.x
        self.y3           = cs.p3.y
        self.x4           = cs.p4.x
        self.y4           = cs.p4.y
        self.T1_x         = cs.T1.x
        self.T1_y         = cs.T1.y
        self.scale_factor = cs.scale_factor.data
